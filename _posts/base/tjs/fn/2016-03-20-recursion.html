            //"use strict";
            /////////////arguments.callee
            function factorial(num){
                if (num<=1){
                    return 1;
                }else{
                    return num*arguments.callee(num-1);//会报错在严格模式下
                }
            }
            var second = factorial;
            factorial = null;
            console.log(second(4));
            

            //////////////////////命名函数，可用于严格模式下
            var anotherfactorial = (function f(num){
                if(num<=1){
                    return 1;
                }else{
                    return num*f(num-1);
                }
            });
            var another = anotherfactorial;
            anotherfactorial = null;
            console.log(another(4));
