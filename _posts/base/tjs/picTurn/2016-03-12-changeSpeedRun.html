            function getStyle(obj,attr){
                return obj.crruentStyle? parseInt(obj.currentStyle[attr]):parseInt(getComputedStyle(obj,false)[attr]);
            }
            window.onload = function(){
                var oDiv = document.getElementById("div1");
                var timer = null;
                var now = 0;
                var speed = 0;
                var target = 100;
                clearInterval(timer);
                timer = setInterval(function(){
                    now = parseInt(oDiv.offsetLeft);
                    
                    if(now == target){
                        clearInterval(timer);
                    }else{
                        speed = speed>0?  Math.ceil((target - now)/6):Math.floor((target - now)/6);
                        oDiv.style.left = now + speed + 'px';
                    }
                },30);
            }
