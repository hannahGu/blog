            tab选项卡 自定义属性在html中加index，不能兼容所有的browser，最好通过js加

            window.onload= function(){
                var oInput= document.getElementsByTagName("input");
                var oContainer = document.getElementById("container");
                var oDiv = oContainer.getElementsByTagName("div");

                for(var i=0;i<oInput.length;i++){
                    oInput[i].index = i;
                    oInput[i].onclick = function(){
                        for(var i=0;i<oInput.length;i++){
                            oInput[i].className = "";
                            oDiv[i].className = "";
                        }
                        this.className = 'active_btn';
                        oDiv[this.index].className='active_div';
                    }
                }
            }

            用闭包解决
            window.onload = function(){
              var oBtn = document.getElementsByTagName("input");
              for(var i=0;i<oBtn.length;i++){
                (function(j){
                  oBtn[j].onclick = function(){
                  for(var i=0;i<oBtn.length;i++){
                    oBtn[i].className="btn"
                  };
                  oBtn[j].className="btn active"
                }
                })(i);
              }
            }
