            js动画之opacity渐变
                opacity:兼容
                opacity:0.3 VS filter:alpha(opacity:30)
                oDiv.style.opacity VS oDiv.style.filter='alpha(opacity:30)'
            //一次mouseout 和一次mouseover都调用一次startMove，startMove前面的判断只执行一次
            //alpha是全局变量,里面保存是的动态更新的opacity值
            function startMove(iTarget){
                    var speed = 0;
                    if(iTarget>alpha){
                        speed = 1;
                    }else{
                        speed = -1;
                    }
                    clearInterval(timer);
                    timer = setInterval(function(){
                        if(alpha == iTarget){
                            clearInterval(timer);
                        }
                        else{
                            alpha += speed;
                            oDiv.style.opacity = alpha/100;
                            oDiv.style.filter = 'alpha(opacity:'+alpha+')';
                        }
                    },30);
                }
