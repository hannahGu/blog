                var oTxt1 = document.getElementById("txt1");
                var oTxt2 = document.getElementById("txt2");
                oTxt1.onkeydown = function(ev){
                    var event = ev || window.event;
                    if(event.keyCode ==13 && event.ctrlKey){
                        oTxt2.value += oTxt1.value + '\n';
                        oTxt1.value = "";
                    }
                    
                }
            }
