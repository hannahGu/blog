            删除某一项（子加事件，删除父）
            window.onload = function(){
                    var aA = document.getElementsByTagName('a');
                    var oUl = document.getElementsByTagName("ul")[0];
                    for(var i=0;i<aA.length;i++){
                        aA[i].onclick = function(){
                            oUl.removeChild(this.parentNode);//从子元素找父元素，删除
                        };
                    }
                }
