            window.onload = function(){
                var oBtn = document.getElementById("button");
                var oTab = document.getElementById("tab1");
                var oTr = oTab.getElementsByTagName('tr');
                oBtn.onclick = function(){
                    var arr = [];
                    for (var i=0;i<oTr.length;i++){
                        arr[i]=oTr[i];
                    }
                    arr.sort(function(l1,l2){
                        var v1 = parseInt(l1.getElementsByTagName('td')[0].innerHTML);
                        var v2 = parseInt(l2.getElementsByTagName('td')[0].innerHTML);
                        return v1 - v2;
                    });
                    for(var i=0;i<arr.length;i++){
                        oTab.appendChild(arr[i]);//oLi[i]对象在整个文档中只有一个。所以不必先删除再append
                    }
                }
            }
