                \d数字  \D除了数字
                *任意字符
                \w英文 数字 下划线   \W除了前面三者
                \s 空白字符  \S

                {n}正好出现n次   {n,}至少n次

                var str = "t4fghrh a 1342ctt345d6 cc 123 ";
                var re = /\d/;
                var re1 =/\d+/g;
                console.log(str.search(re));//1 index
                console.log(str.match(re1));//["4","1342","345","6","123"]//满足的都返回
                console.log(/z/.test(str));//false 检查str中z 字母是否存在
                console.log(str.replace(/\d+/g,'num'));//tnumfghrh a numcttnumdnum cc num 替换后记得赋值
